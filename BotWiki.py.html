import telebot
import wikipedia

wikipedia.set_lang('ru')

TOKEN = 'I WON’t TELL YOU THE NUMBER, FOR I STILL EXERCISE WITH THE BOT. PLEASE FEEL FREE TO CHECK ON THE REST OF THE CODE.'

bot = telebot.TeleBot(TOKEN)

@bot.message_handler(commands=['start'])
def welcome(message):
    bot.send_message(message.chat.id, 'Привет, я Викибот. Напиши название нужной статьи.')

@bot.message_handler(content_types=['text'])
def talk(message):
    if message.text == 'Привет':
        bot.send_message(message.chat.id, 'Привет! Напиши название статьи и я перешлю ее тебе.')
    elif message.text != 'Привет':
        article_name = message.text
        article_name = article_name.replace(" ", "_")
        page = wikipedia.page(article_name)
        bot.send_message(message.chat.id, page.summary)
    else:
        bot.send_message(message.chat.id, 'Упс! Нет такой статьи. Попробуйте написать иначе?')

bot.polling(none_stop=True)
